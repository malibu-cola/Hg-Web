```rust
// 全探索
let mut cnt = 0;
for i in 0..h {
    for j in 0..w {
        if s[i][j] == '#'{continue;}
        let mut ok = true;
        for idx in 0..n {
            // *********
            // 動かす処理
            // *********
            if s[ni][nj] == '#' {
                ok = false;
                break;
            }
        }
        if ok{cnt += 1}
    }
}
```
